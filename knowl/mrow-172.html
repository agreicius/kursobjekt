\begin{align*} \operatorname{area} \amp =\int\limits_\mathcal{C} f\, ds \\ \amp=\int_{-1}^2f(t^2,t)\abs{\boldr'(t)}\, dt \\ \amp = \int_{-1}^2(\sqrt{t^2}+\abs{t})\sqrt{4t^2+1}\, dt\\ \amp = \int_{-1}^2 2\abs{t}\sqrt{4t^2+1}\, dt\\ \amp = \int_{-1}^0-2t\sqrt{4t^2+1}\, dt+\int_0^2 2t\sqrt{4t^2+1}\, dt\\ \amp = \frac{5}{6}\sqrt{5}-\frac{1}{6}+\frac{17}{6}\sqrt{17}-\frac{1}{6}\\ \amp = \frac{17}{6}\sqrt{17}+\frac{5}{6}\sqrt{5}-\frac{1}{3} \end{align*}
