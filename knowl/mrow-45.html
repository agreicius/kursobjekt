\begin{align*} \iint\limits_R x\cos(xy)\, dA \amp=\int_1^2\int_0^\pi x\cos(xy)\, dy\, dx \\ \amp=\int_1^2(\sin(xy))\Bigr\vert_{y=0}^{y=\pi}\, dx\\ \amp = \int_1^2 \sin\pi x \\ \amp = -\frac{1}{\pi}\cos\pi x\Bigr\vert_1^2\\ \amp = -\frac{1}{\pi}(\cos 2\pi-\cos\pi)\\ \amp =-\frac{2}{\pi} \end{align*}
